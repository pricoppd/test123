- name: Retrieve specific key-value pairs from a user secret in Vault
  set_fact:
    specific_key_values: "{{ lookup('hashi_vault', 'secret=user/secrets/johndoe token=' + client_token + ' url=' + host_url + ' namespace=' + hvnamespace) }}"
  with_items:
    - username
    - email
    - role
  register: specific_key_value_results

- name: Output specific key-value pairs
  debug:
    msg: "Key: {{ item.item }}, Value: {{ item.specific_key_values[item.item] }}"
  loop: "{{ specific_key_value_results.results }}"
